Single Linked List

An object is held within a block of memory. 
It is possible for an instance of a class to include a pointer to another instance of the same class

A singly-linked list has a LIFO structure (Last-In, First-Out).

The vector in STL is a singly linked list
Using two classes. Element and LinkedList.
The Element comprises the individual links
The LinkedList is a manager class

Linked List
A sequence of links connected by pointers
End of the list means pointer is at 0.

Example
Class LinkedList
{
     private:
          Element* head;
     public:
          LinkedList::LinkedList()
          {
               // Sets the head to 0 (no elements)
               head = 0;
          }
          LinkedList::~LinkedList()
          {
               Element* prior; // stored previous
               while(head) // 0 at the end
               {             
                    prior = head; // store previous
                    head = head->getNext(); // advance head
                    delete prior; // destroy previous
               }
          }
          void pushBack(int d)
          {
               Element* tmp = new Element(d);
               tmp->setNext(0);
               if(head)
               {
                    // Temporary Variable
                    Element* current = head;
                    // Move all the way till the end
                    while(current->getNext())
                    {
                         current = current->getNext();
                    }
                    // Set the End variable to the new object
                    current->setNext(tmp);
               }
               else
               {
                    // This item is the only one
                    head = tmp;
               }
          }
          void pushFront(int d)
          {
               // Allocate memory for new element
               Element* tmp = new Element(d);
               tmp->setNext(head);
               head = tmp;
          }
          void popFront()
          {
               Element* tmp = head;
               if(head)
               {
                    head = head->getNext();
                    delete tmp;
               }
          }
          void popBack()
          {
               if(head)
               {
                    if(head->getNext() == 0)
                    {
                         // List is empty
                         delete head;
                         head = 0;
                    }
                    else
                    {
                         Element* nextToLast;
                         Element* current = head;
                         while(current->getNext()))
                         {
                              nextToLast = current;
                              current = current->getNext();
                         }
                         nextToLast->setNext(0);
                         delete current;
                    }
               }
          }
          int getBack()
          {          
               if(head)
               {
                    Element* current = head;
                    while(current->getNext())
                    {
                         current = current->getNext();
                    }
                    return current->getData();
               }
          }
          int getFront()
          {
               if(head)
               {
                    return head->getData);
               }
          }
          void display()
          {
               Element* search = head;
               while(search)
               {
                    cout << search->getData();
                    search = search->getNext();
               }
          }
          void pushSort(int d)
          {
               Element* tmp = new Element(d);
               Element* current = head;
               Element* prior = 0;

               while(current)
               {
                    if(current->getData() <= d)
                    {
                         prior = current;
                         current = current->getNext();
                    }
                    else
                    {
                         if(prior)
                         {
                              prior->setNext(tmp);
                              tmp->setNext(current);
                              return;
                         }
                         tmp->setNext(current);
                         head = tmp;
                         return;
                    }
               }
               if(prior)
               {
                    prior->setNext(tmp);
                    tmp->setNext(0);
                    return;
               }
               else
               {
                    tmp->setNext(current);
                    head = tmp;
               }
          }
}
Class Element
{
     private:
          Element* next;
          int data;
     public:
          int getData();
          {
               return data;
          }
          void setNext(Element* tmp);
          {
               next = tmp;
          }
          Element* getNext()
          {
               return next;
          }
          Element::Element(int d)
          {
               data = d;
          }
}
